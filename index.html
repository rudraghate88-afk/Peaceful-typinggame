// Central Typing Processor Core (Shared between PC & Mobile) function handleInputCharacter(inputKey) { if (charIndex >= currentText.length) return; if (inputKey === 'Backspace') { if (charIndex > 0) { charIndex--; const prevSpan = document.getElementById(`char-${charIndex}`); if (prevSpan) prevSpan.classList.remove('correct', 'incorrect'); updateCursor(); } return; } if (inputKey.length !== 1) return; const targetChar = currentText[charIndex]; const currentSpan = document.getElementById(`char-${charIndex}`); if (currentSpan) { if (inputKey === targetChar) { currentSpan.classList.add('correct'); currentSpan.classList.remove('incorrect'); createSparkles(); playHeavenlyChime(); } else { currentSpan.classList.add('incorrect'); } } charIndex++; if (charIndex <= currentText.length) { updateCursor(); } } // Desktop Listener window.addEventListener('keydown', (e) => { initAudio(); // Activate sound tracking context if (e.key === ' ' || e.key === 'Spacebar') { e.preventDefault(); } handleInputCharacter(e.key); }); // Mobile Target Automation Fixes (Unclogged from Predictive Text) const mobileInput = document.getElementById('mobile-keyboard-trigger'); document.addEventListener('click', () => { initAudio(); if (mobileInput) { mobileInput.focus(); } }); if (mobileInput) { mobileInput.addEventListener('input', (e) => { // If it's a backspace action if (e.inputType === 'deleteContentBackward') { handleInputCharacter('Backspace'); mobileInput.value = ''; return; } // Read ONLY the very last character actually typed into the box, ignoring auto-correct blocks const val = mobileInput.value; if (val.length > 0) { const lastChar = val.charAt(val.length - 1); handleInputCharacter(lastChar); } mobileInput.value = ''; // Instantly clear to keep auto-correct fresh }); } window.onload = () => { loadPrompt('dance'); };
